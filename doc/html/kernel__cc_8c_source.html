61 /*
62  Pre-emption aware mutex.
63  -------------------------
64 
65  This mutex will act as a spinlock if preemption is off, and a
66  yielding mutex if it is on.
67 
68  */
69 void Mutex_Lock(Mutex* lock)
70 {
71 #define MUTEX_SPINS 1000
72 
73  int spin=MUTEX_SPINS;
74  while(__atomic_test_and_set(lock,__ATOMIC_ACQUIRE)) {
75  while(__atomic_load_n(lock, __ATOMIC_RELAXED)) {
76  __builtin_ia32_pause();
77  if(spin>0)
78  spin--;
79  else {
80  spin=MUTEX_SPINS;
81  if(get_core_preemption())
82  yield();
83  }
84  }
85  }
86 #undef MUTEX_SPINS
87 }
